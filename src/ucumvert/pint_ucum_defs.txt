# Units definition file to extend pints default.txt with UCUM units
# Language: english
# Syntax: https://pint.readthedocs.io/en/stable/advanced/defining.html

# Pint unit names, aliases and prefixes have to be valid python identifiers.
# So UCUM units such as "10*", "10^" or "[gal_us]" are invalid unit names in pint.
# We handle this as follows:
# - If the unit can be unambiguously mapped we just define an alias dropping the brackets.
# - Else we define a unit using "_" as prefix. Operators like "*", "^" are dropped.
# Example: [gal_us] -> Alias: "gal_us"
#          [10*] -> Unit: "_10"

#### Prefixes ####
# <prefix>- = <amount> [= <symbol>] [= <alias>] [ = <alias> ] [...]
# Example:
#     deca- =  1e+1  = da- = deka-

#### UNITS ####
# <canonical name> = <relation to another unit or dimension> [= <symbol>] [= <alias>] [ = <alias> ] [...]
# The canonical name and aliases should be expressed in singular form.
# If a unit has no symbol, then the symbol should be set to _.
# Example:
#     millennium = 1e3 * year = _ = millennia

ppth = 1e-3
ppb = 1e-9
pptr = 1e-12

international_unit = 1 = i.U. = IU = iU
_10 = 10 = _ = _10*
equivalents = mol = eq
osmole = mol = osm

#### Additional aliases ####
# @alias <canonical name or previous alias> = <alias> [ = <alias> ] [...]

@alias are = ar = AR
@alias degree_Celsius = Cel = CEL
@alias millimeter_Hg = mm[Hg] = mM[HG]

@alias week = wk = WK
@alias month = mo = MO

@alias ohm = Ohm = OHM

@alias international_feet = ft_i = FT_I
@alias survey_foot = ft_us = FT_US

@alias gallon = gal_us = GAL_US
@alias oil_barrel = bbl_us = BBL_US
